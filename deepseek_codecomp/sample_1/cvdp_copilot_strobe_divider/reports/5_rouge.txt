task apply_strobe_divider_with_out_ready;
    input integer latency;
    input integer cycles;
    input integer deassert_at;
    integer k;
    begin
        for (k = 0; k < cycles; k = k + 1) begin
            @(posedge (latency == 0 ? Clk_0 : Clk_1));
            if (k == deassert_at) begin
                if (latency == 0) begin
                    Out_Ready_0 <= 0; 
                end else begin
                    Out_Ready_1 <= 0; 
                end
            end else begin
                if (latency == 0) begin
                    Out_Ready_0 <= 1; 
                end else begin
                    Out_Ready_1 <= 1; 
                end
            end

            // Toggle In_Valid
            if (k % 2 == 0) begin
                if (latency == 0) begin
                    In_Valid_0 <= 1;
                end else begin
                    In_Valid_1 <= 1;
                end
            end else begin
                if (latency == 0) begin
                    In_Valid_0 <= 0;
                end else begin
                    In_Valid_1 <= 0;
                end
            end
        end
    end
endtask
