    reg [7:0] count;			    // General purpose counter to sync with bit_count


    // Sequential logic block to capture serial data on the rising edge of serial_clk
    always @(posedge serial_clk or negedge  reset_n) begin
        if (!reset_n) begin		    // Reset all registers on reset_n (active low)
            data_reg <= 64'h0;		// Clear the data register
            bit_count <= 8'd0;		// Clear the bit counter
        end else begin

                bit_count <= 8'b0;
                data_reg <= 64'h0;

                 bit_count <= bit_count;			 // Maintain the bit counter if not done
