    input  wire                   rst_n,
    input  wire [DATA_WIDTH-1:0] data_src,
    input  wire                   valid_src,

    output reg  [DATA_WIDTH-1:0] data_dst,
    output reg                   valid_dst
);

    reg [DATA_WIDTH-1:0] sync1_data, sync2_data;
    reg                  sync1_valid; 

    always @(posedge clk_dst or negedge rst_n) begin

            sync1_data  <= 0; sync2_data <= 0;

            data_dst  <= 0;

module lfsr_generator #(
    parameter WIDTH = 16,     
    parameter SEED = 16'hACE1
)(
    input  wire clk,
    input  wire rst_n,
    output reg [WIDTH-1:0] lfsr
);

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            lfsr <= SEED;
        else
            lfsr <= {lfsr[WIDTH-2:0], lfsr[WIDTH-1] ^ lfsr[2]};
    end

endmodule
